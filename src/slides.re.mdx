# **Three.js**

### Renderer

```js
import * as THREE from 'three';

const scene = new THREE.Scene();
const aspect = window.innerWidth / window.innerHeight;
const camera = new THREE.PerspectiveCamera(75, aspect, 0.1, 1000);
const renderer = new THREE.WebGLRenderer();
renderer.setSize(window.innerWidth, window.innerHeight);

document.body.appendChild(renderer.domElement);
```

### Escena

```js
const geometry = new THREE.SphereGeometry(1, 18, 18);
const material = new THREE.MeshNormalMaterial();
const sphere = new THREE.Mesh(geometry, material);

scene.add(sphere);
camera.position.z = 3;
```

### Ciclo principal

```js
const render = () => {
  requestAnimationFrame(render);
  sphere.rotation.y += 0.1;
  renderer.render(scene, camera);
};

render();
```

# **React Three Fiber**

```jsx
import { Canvas, useFrame } from "@react-three/fiber";
import { useRef } from "react";

const Sphere = () => {
  const ref = useRef(null);

  useFrame((_state, delta) => {
    ref.current.rotation.y += delta;
  });

  return (
    <mesh ref={ref} position={[0, 0, 0]}>
      <sphereGeometry args={[1, 18, 18]} />
      <meshStandardMaterial color="blue" />
    </mesh>
  );
};

export const Example = () => {
  return (
    <Canvas camera={{ position: [0, 0, 3] }}>
      <ambientLight />
      <Sphere />
    </Canvas>
  );
};
```

# Material

```jsx
import { Canvas, useFrame, useLoader } from "@react-three/fiber";
import { useRef } from "react";
import { TextureLoader } from "three";

const Earth = () => {
  const ref = useRef<(null);
  const colorMap = useLoader(TextureLoader, "/images/earth.jpeg");

  useFrame((_state, delta) => {
    ref.current.rotation.y += delta;
  });

  return (
    <mesh ref={ref} position={[0, 0, 0]}>
      <sphereGeometry args={[1, 18, 18]} />
      <meshStandardMaterial map={colorMap} />
    </mesh>
  );
};

export const Example = () => {
  return (
    <Canvas>
      <Suspense fallback={null}>
        <ambientLight />
        <Earth />
      </Suspense>
    </Canvas>
  );
};
```

# Iluminación

```jsx
const Sun = () => {
  const ref = useRef(null);
  const position = [0, 0, 10]

  useFrame((_state, delta) => {
    ref.current.rotation.y += delta * 2;
  });

  return (
    <group ref={ref}>
      <pointLight position={position} />
      <mesh position={position}>
        <sphereGeometry args={[0.2, 18, 18]} />
        <meshStandardMaterial color="yellow" />
      </mesh>
    </group>
  );
};

export const LightExample = () => {
  return (
    <Canvas camera={{ position: [0, 0, 3] }}>
      <Suspense fallback={null}>
        <ambientLight />
        <Sun position={[0, 0, 0]} />
        <Earth position={[0, 0, 0]} />
      </Suspense>
    </Canvas>
  );
};
```

# Modelo

```jsx
import { Canvas, useLoader } from "@react-three/fiber";
import { OrbitControls } from "@react-three/drei";
import { GLTFLoader } from "three/addons/loaders/GLTFLoader.js";

export const FlatExample = () => {
  const gltf = useLoader(GLTFLoader, "/assets/flat.glb");

  return (
    <Canvas resize={{ scroll: false, offsetSize: true }}>
      <ambientLight intensity={0.5} />
      <pointLight position={[-10, -10, -10]} />
      <primitive object={gltf.scene} />
      <OrbitControls />
    </Canvas>
  );
};
```
